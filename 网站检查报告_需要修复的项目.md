## ❌ 问题 1: 俄语页面 Meta 标签未优化

### 当前状态

**文件位置**: `/aierxuan-website/src/app/[lang]/layout.tsx` (第14-33行)

```typescript
export async function generateMetadata({ params }: { params: Promise<{ lang: string }> }): Promise<Metadata> {
  const { lang } = await params

  return {
    title: 'AIERXUAN - Professional Laptop & Mini PC Manufacturer',  // ❌ 英文
    description: 'Leading OEM/ODM manufacturer of high-performance laptops, gaming notebooks, and mini PCs. Custom solutions for global partners.',  // ❌ 英文
    keywords: 'laptop manufacturer, mini pc factory, oem laptop, odm notebook, gaming laptop supplier, shenzhen electronics',  // ❌ 英文
    // ...
  }
}
```

### 问题说明

- 所有语言（英语、俄语）都使用相同的英文 Meta 标签
- 这严重影响俄语页面在 Yandex 的搜索排名
- 俄语用户搜索时看到的是英文标题和描述

### 修复方案

需要根据 `lang` 参数返回不同语言的 Meta 标签。

#### 修复代码

```typescript
export async function generateMetadata({ params }: { params: Promise<{ lang: string }> }): Promise<Metadata> {
  const { lang } = await params

  // 根据语言返回不同的 Meta 标签
  const metaContent = {
    en: {
      title: 'AIERXUAN - Professional Laptop & Mini PC Manufacturer',
      description: 'Leading OEM/ODM manufacturer of high-performance laptops, gaming notebooks, and mini PCs. Custom solutions for global partners.',
      keywords: 'laptop manufacturer, mini pc factory, oem laptop, odm notebook, gaming laptop supplier, shenzhen electronics',
    },
    ru: {
      title: 'AIERXUAN - Производитель Ноутбуков и Мини-ПК OEM/ODM | Россия',
      description: 'Профессиональный производитель ноутбуков и мини-ПК для российского рынка. OEM/ODM решения от 1 образца до 10000+ единиц. Intel Partner, сертификация CE/FCC. Доставка 7-15 дней. Telegram: @aierxuan_russia',
      keywords: 'производитель ноутбуков, OEM ноутбуки, ODM мини ПК, купить ноутбуки оптом Россия, бизнес ноутбуки, игровые ноутбуки, мини компьютеры, Intel партнер Россия, производство ноутбуков на заказ',
    },
  }

  const content = metaContent[lang as keyof typeof metaContent] || metaContent.en

  return {
    title: content.title,
    description: content.description,
    keywords: content.keywords,
    icons: {
      icon: '/aierxuan_favicon.svg',  // 修复 favicon 路径
    },
    alternates: {
      canonical: `${BASE_URL}/${lang}`,
      languages: {
        'x-default': `${BASE_URL}/en`,
        'en': `${BASE_URL}/en`,
        'ru': `${BASE_URL}/ru`,
      },
    },
    openGraph: {
      title: content.title,
      description: content.description,
      locale: lang === 'ru' ? 'ru_RU' : 'en_US',
      type: 'website',
      url: `${BASE_URL}/${lang}`,
    },
  }
}
```

## ❌ 问题 2: Yandex Metrica 未安装

### 当前状态

- 只安装了 Google Tag Manager
- 没有 Yandex Metrica 代码
- 无法追踪俄罗斯用户行为数据

### 修复方案

需要在 layout.tsx 中添加 Yandex Metrica 代码。

#### 步骤 1: 获取 Yandex Metrica ID

首先需要：
1. 登录 Yandex Metrica: https://metrica.yandex.com
2. 创建新的计数器（Counter）
3. 获取计数器 ID（例如：12345678）

#### 步骤 2: 添加环境变量

在 `.env.local` 文件中添加：

```bash
NEXT_PUBLIC_YANDEX_METRICA_ID=YOUR_COUNTER_ID
```

#### 步骤 3: 修改 layout.tsx

在现有的 Google Tag Manager 代码之后添加：

```typescript

                clickmap:true,
```

## 📝 完整修复后的 layout.tsx 代码

```typescript
import type { Metadata } from 'next'
import Script from 'next/script'
import { ConditionalNavbar } from '@/components/layout/ConditionalNavbar'
import { Footer } from '@/components/layout/Footer'
import { ContactModalLazy } from '@/components/ui/ContactModalLazy'
import { FloatingCTAButtonLazy } from '@/components/ui/FloatingCTAButtonLazy'
import { TelegramFloatingButton } from '@/components/ui/TelegramFloatingButton'
import { HtmlLangSetter } from '@/components/layout/HtmlLangSetter'
import { i18n, type Locale } from '@/i18n-config'
import { getDictionary } from '@/get-dictionary'

const BASE_URL = 'https://aierxuanlaptop.com'

export async function generateMetadata({ params }: { params: Promise<{ lang: string }> }): Promise<Metadata> {
  const { lang } = await params

  // 多语言 Meta 内容
  const metaContent = {
    en: {
      title: 'AIERXUAN - Professional Laptop & Mini PC Manufacturer',
      description: 'Leading OEM/ODM manufacturer of high-performance laptops, gaming notebooks, and mini PCs. Custom solutions for global partners.',
      keywords: 'laptop manufacturer, mini pc factory, oem laptop, odm notebook, gaming laptop supplier, shenzhen electronics',
    },
    ru: {
      title: 'AIERXUAN - Производитель Ноутбуков и Мини-ПК OEM/ODM | Россия',
      description: 'Профессиональный производитель ноутбуков и мини-ПК для российского рынка. OEM/ODM решения от 1 образца до 10000+ единиц. Intel Partner, сертификация CE/FCC. Доставка 7-15 дней. Telegram: @aierxuan_russia',
      keywords: 'производитель ноутбуков, OEM ноутбуки, ODM мини ПК, купить ноутбуки оптом Россия, бизнес ноутбуки, игровые ноутбуки, мини компьютеры, Intel партнер Россия, производство ноутбуков на заказ',
    },
  }

  const content = metaContent[lang as keyof typeof metaContent] || metaContent.en

  return {
    title: content.title,
    description: content.description,
    keywords: content.keywords,
    icons: {
      icon: [
        { url: '/aierxuan_favicon.svg', type: 'image/svg+xml' },
        { url: '/aierxuan_favicon.png', sizes: '512x512', type: 'image/png' },
      ],
      apple: '/aierxuan_favicon.png',
    },
    alternates: {
      canonical: `${BASE_URL}/${lang}`,
      languages: {
        'x-default': `${BASE_URL}/en`,
        'en': `${BASE_URL}/en`,
        'ru': `${BASE_URL}/ru`,
      },
    },
    openGraph: {
      title: content.title,
      description: content.description,
      locale: lang === 'ru' ? 'ru_RU' : 'en_US',
      type: 'website',
      url: `${BASE_URL}/${lang}`,
    },
  }
}

export async function generateStaticParams() {
  return i18n.locales.map((locale) => ({ lang: locale }))
}

export default async function RootLayout({
  children,
  params,
}: {
  children: React.ReactNode
  params: Promise<{ lang: string }>
}) {
  const { lang: langParam } = await params
  const lang = langParam as Locale
  const dictionary = await getDictionary(lang)
  const gtmId = process.env.NEXT_PUBLIC_GTM_ID
  const ymId = process.env.NEXT_PUBLIC_YANDEX_METRICA_ID

  return (
    <>
      <HtmlLangSetter lang={lang} />

      {/* Google Tag Manager */}
      {gtmId ? (
        <>
          <Script
            id="gtm-script"
            strategy="lazyOnload"
            dangerouslySetInnerHTML={{
              __html: `(function(w,d,s,l,i){w[l]=w[l]||[];w[l].push({'gtm.start':new Date().getTime(),event:'gtm.js'});var f=d.getElementsByTagName(s)[0],j=d.createElement(s),dl=l!='dataLayer'?'&l='+l:'';j.async=true;j.src='https://www.googletagmanager.com/gtm.js?id='+i+dl;f.parentNode.insertBefore(j,f);})(window,document,'script','dataLayer','${gtmId}');`,
            }}
          />
          <noscript>
            <iframe
              src={`https://www.googletagmanager.com/ns.html?id=${gtmId}`}
              height="0"
              width="0"
              style={{ display: 'none', visibility: 'hidden' }}
            />
          </noscript>
        </>
      ) : null}

      {/* Yandex Metrica */}
      {ymId ? (
        <Script
          id="yandex-metrica"
          strategy="afterInteractive"
          dangerouslySetInnerHTML={{
            __html: `
              (function(m,e,t,r,i,k,a){
                m[i]=m[i]||function(){(m[i].a=m[i].a||[]).push(arguments)};
                m[i].l=1*new Date();
                for (var j = 0; j < document.scripts.length; j++) {
                  if (document.scripts[j].src === r) { return; }
                }
                k=e.createElement(t),a=e.getElementsByTagName(t)[0],k.async=1,k.src=r,a.parentNode.insertBefore(k,a)
              })(window, document, "script", "https://mc.yandex.ru/metrika/tag.js", "ym");

              ym(${ymId}, "init", {
                clickmap:true,
                trackLinks:true,
                accurateTrackBounce:true,
                webvisor:true
              });
            `,
          }}
        />
      ) : null}

      <div className="min-h-screen flex flex-col">
        <ConditionalNavbar dictionary={dictionary} lang={lang} />
        <main className="flex-1">
          {children}
        </main>
        <Footer dictionary={dictionary.footer} lang={lang} />
      </div>

      {/* Global Contact Modal */}
      <ContactModalLazy />

      {/* Floating CTA Button */}
      <FloatingCTAButtonLazy />

      {/* Telegram Floating Button - 仅俄语页面显示 */}
      <TelegramFloatingButton />
    </>
  );
}
```
